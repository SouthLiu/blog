# MySQL
### 配置数据库：
```python
  DATABASES = {
    'default': {
      # 配置数据库引擎 ， 使用 MySQL 数据库
      'ENGINE': 'django.db.backends.mysql',
      'HOST' : '127.0.0.1',
      'USER' : 'root',
      'PASSWORD' : 'root',
      # 配置连接的数据库 ， 数据库必须先创建 ， 才可以进行链接
      'NAME' : 'django_orm'
    }
  }
```

### 获取数据：
```python
from django.http import HttpResponse
from django.db import connection

def index(request):
    # 获取游标对象
    cursor = connection.cursor()
    # 执行SQL语句
    cursor.execute("SELECT * FROM book")
    # 获取查询结果
    result = cursor.fetchall()
    # 遍历查询结果
    for row in result:
        print(row)
    return HttpResponse("打印查询数据.")
```

# ORM模型
### 通过ORM创建数据库：
```python
# book/models.py
from django.db import models

class Book(models.Model):
    title = models.CharField(max_length=100)
    author = models.CharField(max_length=100)
    genre = models.CharField(max_length=100)
    publication_date = models.DateField(auto_now_add=True)
```

### 数据过滤：
```python
# book/views.py
from .models import Book

def get_books(request):
    # 查询所有书籍
    books = Book.objects.all()
    # 查询特定书籍
    books = Book.objects.filter(title='Python入门')

    for book in books:
        print(book.title, book.author, book.genre, book.publication_date)
    return HttpResponse("打印查询数据.")
```

### 获取单个对象：
```python
# book/views.py
from .models import Book

def get_books(request):
    try:
        # 获取单个对象
        book = Book.objects.get(id=2)
        print(book.title, book.author, book.genre, book.publication_date)
    except Book.DoesNotExist:
        print("查询数据不存在.")
    return HttpResponse("打印查询数据.")
```

### 排序：
```python
# book/views.py
from .models import Book

def order_books(request):
    books = Book.objects.order_by('publication_date') # 正序
    # books = Book.objects.order_by('-publication_date') # 倒序
    for book in books:
        print(book.title, book.author, book.genre)
    return HttpResponse("查询所有排序图书成功.")
```

### 更新数据：
```python
# book/views.py
from .models import Book

def update_book(request):
    try:
        # 更新单个对象
        book = Book.objects.get(id=2)
        book.title = 'Python入门-基础'
        book.save()
    except Book.DoesNotExist:
        print("查询数据不存在.")
    return HttpResponse("更新图书成功.")
```

### 删除数据：
```python
# book/views.py
from .models import Book

def delete_book(request):
    try:
        # 删除单个对象
        book = Book.objects.get(id=2)
        book.delete()
    except Book.DoesNotExist:
        print("查询数据不存在.")
    return HttpResponse("删除图书成功.")
```
