# 冒泡排序
```cpp
#include <iostream>
using namespace std;

// 冒泡排序
void bubbleSort(int *num, int len) {
	for (int i = 0; i < len; i++) {
		for (int j = 0; j < len - i - 1; j++) {
			if (num[j] > num[j + 1]) {
				int temp = num[j];
				num[j] = num[j + 1];
				num[j + 1] = temp;
			}
		}
	}
}

// 打印数组
void print(int* arr, int len) {
	for (int i = 0; i < len; i++) {
		cout << i << ":" << arr[i] << endl;
	}
}

int main() {
	int arr[] = { 9, 3, 5, 1 };
	int len = sizeof(arr) / sizeof(arr[0]);
	bubbleSort(arr, len);
	print(arr, len);

	return 0;
}
```
