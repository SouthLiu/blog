# 冒泡排序
```cpp
#include <iostream>
using namespace std;

// 冒泡排序
void bubbleSort(int *num, int len) {
	for (int i = 0; i < len; i++) {
		for (int j = 0; j < len - i - 1; j++) {
			if (num[j] > num[j + 1]) {
				int temp = num[j];
				num[j] = num[j + 1];
				num[j + 1] = temp;
			}
		}
	}
}

// 打印数组
void print(int* arr, int len) {
	for (int i = 0; i < len; i++) {
		cout << i << ":" << arr[i] << endl;
	}
}

int main() {
	int arr[] = { 9, 3, 5, 1 };
	int len = sizeof(arr) / sizeof(arr[0]);
	bubbleSort(arr, len);
	print(arr, len);

	return 0;
}
```

# 结构体案例
```cpp
#include <iostream>
using namespace std;
#include <string>
#include <ctime>

struct student {
	string name;
	int score;
};

struct teacher {
	string name;
	student stu[5];
};

// 提供数据
void provideData(teacher tearchers[], int len) {
	string symbol = "ABCDE";
	for (int i = 0; i < len; i++) {
		tearchers[i].name = "老师";
		tearchers[i].name += symbol[i];

		for (int j = 0; j < 5; j++) {
			tearchers[i].stu[j].name = "学生";
			tearchers[i].stu[j].name += symbol[j];
			tearchers[i].stu[j].score = rand() % 61 + 40;
		}
	}
}

// 打印全部信息
void print(teacher data[], int len) {
	for (int i = 0; i < len; i++) {
		cout << "教师名称：" << data[i].name << endl;

		for (int j = 0; j < 5; j++) {
			cout << "  学生名称：" << data[i].stu[j].name
				<< " 学生分数：" << data[i].stu[j].score << endl;
		}
	}
}

void main() {
	// 随机数种子，不添加每次启动的随机数都相同
	srand((unsigned int)time(NULL);
	teacher teachers[3];
	int len = sizeof(teachers) / sizeof(teachers[0]);
	provideData(teachers, len);
	print(teachers, len);
}
```

# 根据年龄升序
```cpp
#include <iostream>
using namespace std;
#include <string>

struct Person {
	string name;
	int age;
	string sex;
};

// 根据年龄升序
void ageSort(Person persons[], int len) {
	for (int i = 0; i < len; i++) {
		for (int j = 0; j < len - i - 1; j++) {
			if (persons[j].age > persons[j + 1].age) {
				Person temp = persons[j];
				persons[j] = persons[j + 1];
				persons[j + 1] = temp;
			}
		}
	}
}

// 打印数据
void print(Person persons[], int len) {
	for (int i = 0; i < len; i++) {
		cout << "名字：" << persons[i].name
			<< " 年龄：" << persons[i].age
			<< " 性别：" << persons[i].sex << endl;
	}
}

void main() {
	Person persons[] = {
		{"刘备", 18, "男"},
		{"关羽", 22, "男"},
		{"张飞", 20, "男"},
		{"貂蝉", 19, "女"},
		{"小乔", 16, "女"}
	};
	int len = sizeof(persons) / sizeof(persons[0]);
	ageSort(persons, len);
	print(persons, len);
}
```

# 求立方体面积和体积，且对比两个立方体是否相同
```cpp
#include <iostream>
using namespace std;
#include <string>

class Cube {
public:
	void setCube(int height, int weight, int length) {
		h = height;
		w = weight;
		l = length;
	};

	// 获取长
	int getWeight() {
		return w;
	};

	// 获取宽
	int getLength() {
		return l;
	};

	// 获取高
	int getHeight() {
		return h;
	};

	// 获取面积
	int getArea() {
		if (!h || !w || !l) {
			cout << "不是一个合法的立方体" << endl;
			return 0;
		};
		return 2 * (h * w + h * l + w * l);
	};

	// 获取体积
	int getVolume() {
		if (!h || !w || !l) {
			cout << "不是一个合法的立方体" << endl;
			return 0;
		};
		return h * w * l;
	};

	// 对比两个立方体是否相等
	bool isSame(Cube& c) {
		if (c.getWeight() == w && c.getHeight() == h && c.getLength() == l) {
			return true;
		}
		return false;
	};

private:
	int h = 0;
	int w = 0;
	int l = 0;
};

void main() {
	Cube cube;
	cube.setCube(10, 20, 30);
	cout << cube.getArea() << endl;
	cout << cube.getVolume() << endl;

	// 创建第二个立方体
	Cube c2;
	c2.setCube(10, 20, 30);
	bool res = cube.isSame(c2);
	string resStr = res ? "相同" : "不同";
	cout << "两个立方体" << resStr << endl;
	system("pause");
}
```
